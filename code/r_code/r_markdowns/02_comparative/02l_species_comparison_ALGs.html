Comparison

We will use comparABle’s implementation of the Jensen-Shannon Distance:

# Quant norm
a <- qnorm(a)
## Loading required package: preprocessCore
b <- qnorm(b)
c <- qnorm(c)

# The correlations:
cors_ab <- rawcorsp(a_o = a, b_o = b)
cors_ac <- rawcorsp(a_o = a, b_o = c)
cors_bc <- rawcorsp(a_o = b, b_o = c)

Then the heatmaps:

h_ab <-
  Heatmap(
    name = "JSD Pf/Sp",
    cors_ab$js,
    cluster_rows = F, cluster_columns = F, show_row_names = F, show_column_names = F,
    col = brewer.pal(10,"RdBu"),
    left_annotation = devstages_ha_rows(),
    top_annotation = quick_ha(colnames(b),"Purple-Orange",rev = TRUE)

h_ac <-

draw(h_ab+h_ac)

Pearson correlation at once of all the species:

Since the JSD metric varies a lot between the species it does not make sense to normalise there. We can however use the metric of Spearman correlation as a proxy to see which stages are most similar between species.

abc <- cbind(a,b,c)

abc_sp <- cor(abc, method = "sp")
ha = 
  HeatmapAnnotation(
    stage = colnames(abc),
    col = list(stage = 
                 setNames(
                   c(
                     dev_palette,
                     sequential_hcl(ncol(b),"Purple-Orange",rev = TRUE),
                     sequential_hcl(ncol(c),"SunsetDark",rev = TRUE)
                   ),
                   colnames(abc)
                 )),
    show_legend = FALSE, which = "column"
  )

ha_row = 
  HeatmapAnnotation(
    stage = colnames(abc),
    col = list(stage = 
                 setNames(
                   c(
                     dev_palette,
                     sequential_hcl(ncol(b),"Purple-Orange",rev = TRUE),
                     sequential_hcl(ncol(c),"SunsetDark",rev = TRUE)
                   ),
                   colnames(abc)
                 )),
    show_legend = FALSE, which = "row"
  )

h_abc_sp <- 
  Heatmap(
    name = "Spearman",
    abc_sp,
    col = colorRamp2(seq(0,0.75, length = 10),rev(brewer.pal(10,"RdYlBu"))),
    cluster_rows = FALSE,
    cluster_columns = FALSE,
    row_split = factor(c(rep("PF",ncol(a)),rep("SP",ncol(b)),rep("BF",ncol(c))), levels = c("PF","SP","BF")),
    column_split = factor(c(rep("PF",ncol(a)),rep("SP",ncol(b)),rep("BF",ncol(c))), levels = c("PF","SP","BF")),
    row_names_gp = gpar(cex = .5),
    column_names_gp = gpar(cex = .5),
    top_annotation = ha,
    right_annotation = ha_row,
    bottom_annotation = ha,
    left_annotation = ha_row
  )


draw(h_abc_sp)
